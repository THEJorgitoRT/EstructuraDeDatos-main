#include <bits/stdc++.h>
using namespace std;

void burbujasimple(std::vector<std::string> &arr){
    int n = arr.size();

    for (int i = 0; i < n - 1; i++){
        for (int j = 0; j < n - i - 1; j++){
            if (arr[j] > arr[j+1]) {
                string temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }
}

import java.util.*;
import java.lang.*;
import java.io.*;

class BurbujaSimple
{
    
    public static void OrdenamientoBurbuja(String [] arr){
        int n = arr.length;
        for (int i= 0; i < n - 1; i++){
            for (int j=0;j < n - i - 1; j++){
                if (arr[j].compareTo(arr[j+1]) > 0){
                    // Intercambiar Arr [j y arr [i]]
                    String temp = arr[j];
                    arr[j] = arr[i + 1];
                    arr[i+1] = temp;
                } 
            }
        }
    }
    public static void main (String[] args) throws java.lang.Exception
    {
        String[] nombres = {"carlos", "ana", "luis", "Maria",  "pedro"};
        System.out.println("Lista Original: ");
        for (String nombre : nombres){
            System.out.print(nombre + " ");
        }
        OrdenamientoBurbuja(nombres);
        System.out.println("\nLista Ordenada: ");
        for (String nombre : nombres){
            System.out.print(nombre + " ");
        }
    }
}

import java.util.*;
import java.lang.*;
import java.io.*;

class BurbujaMejorado
{
	public static void main (String[] args) throws java.lang.Exception
	{
		int vec[] = {47,17,23,67,21};
		System.out.println("Vector Original: ");
	    imprimirVector(vec);
		System.out.println("");
		vec = ordenacionBurMej(vec);
		System.out.println("Vector Ordenado: ");
		imprimirVector(vec);

	}
	public static int [] ordenacionBurMej(int vec[]){
	    final int N = vec.length;
	    for (int i = N - 1;i > 0 ; i-- ){
	        for (int j = 0;j < i ; j++ ){
	            if (vec[j]>vec[j+1]){
	                int temp = vec[j];
	                vec[j] = vec[j+1];
	                vec[j+1] = temp;
	            } 
	        } System.out.print("Iteracion: "+ (N-i)+": ");
	        imprimirVector(vec);
	        System.out.println();
	    } 
	    return vec;
	}
	public static void imprimirVector(int vec[]){
	    for (int i = 0;i < vec.length ;i++ ){
	        System.out.print(vec[i] + " ");
	    } 
	}
}
