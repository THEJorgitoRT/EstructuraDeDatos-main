#include <bits/stdc++.h>
using namespace std;

void burbujasimple(std::vector<std::string> &arr){
    int n = arr.size();

    for (int i = 0; i < n - 1; i++){
        for (int j = 0; j < n - i - 1; j++){
            if (arr[j] > arr[j+1]) {
                string temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }
}
